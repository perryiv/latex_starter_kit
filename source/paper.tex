% Define what kind of document this is.
\documentclass[twocolumn]{article}

% For including images and eps files.
\usepackage{graphics}
\usepackage{epsfig}

% Set the size of the paper, margins, etc.
\usepackage{geometry}
\geometry{letterpaper, portrait, margin=0.5in}
\setlength{\columnsep}{0.4in}

% Automatically generate the bibliography.
\usepackage[
  sortcites = true,        % \cite{a,d,b} --> [a,b,d]
  citestyle = numeric-comp % \cite{a,c,b} --> [a-c]
]{biblatex}
\addbibresource{references.bib}

% Set the title and author.
\title{\LARGE \bf My Paper Title}
\author{My Name}

% Start of the document.
\begin{document}

% Define some variables used throughout.
\include{definitions}

% Make the title
\maketitle

% No page number on the very first page.
\thispagestyle{empty}

% Include the content.
\input{abstract}
\input{introduction}
\input{discussion}
\input{conclusion}
\input{appendix}
\input{acknowledgments}

% Generate the bibliography.
\printbibliography

% End of the document.
\end{document}
